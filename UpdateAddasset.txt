const ALLOWED_IMAGE_TYPES = ["image/png", "image/jpg", "image/jpeg"];
const MAX_FILE_SIZE_MB = 5;

// 2. Update updateForm to validate files immediately on selection
const updateForm = function (name: string, value: any) {
    // File validation for certificate fields
    if (
        (name === "oem_certificate" || name === "third_party_certificate") &&
        value instanceof File
    ) {
        if (!ALLOWED_IMAGE_TYPES.includes(value.type)) {
            setErrors((prev: any) => ({
                ...prev,
                [name]: "Only PNG, JPG, or JPEG images are allowed",
            }));
            return; // Don't update formData if invalid
        }
        if (value.size > MAX_FILE_SIZE_MB * 1024 * 1024) {
            setErrors((prev: any) => ({
                ...prev,
                [name]: `File size must not exceed ${MAX_FILE_SIZE_MB}MB`,
            }));
            return; // Don't update formData if invalid
        }
    }

    setFormData((prev) => ({
        ...prev,
        [name]: value,
    }));
    if (errors[name]) {
        setErrors((prev: any) => ({ ...prev, [name]: "" }));
    }
};

// Dependency rules: when key permission is selected, value permissions are auto-selected & locked.
const PERMISSION_DEPENDENCIES: Record<string, { module: string; action: string }[]> = {
    "user:create": [{ module: "role", action: "list" }],
};

/** Returns true if this permission is force-selected due to an active dependency. */
const isPermissionLocked = (module: string, action: string): boolean => {
    return Object.entries(PERMISSION_DEPENDENCIES).some(([key, deps]) => {
        const [depModule, depAction] = key.split(":");
        const isSourceActive = selectedPermissions[depModule]?.includes(depAction);
        return isSourceActive && deps.some((d) => d.module === module && d.action === action);
    });
};

/** Applies or removes dependency side-effects when a permission changes. */
const applyDependencies = (
    prev: Record<string, string[]>,
    module: string,
    action: string,
    isAdding: boolean
): Record<string, string[]> => {
    const key = `${module}:${action}`;
    const deps = PERMISSION_DEPENDENCIES[key];
    if (!deps) return prev;

    let updated = { ...prev };

    if (isAdding) {
        // Auto-add all dependent permissions
        deps.forEach(({ module: dMod, action: dAct }) => {
            const existing = updated[dMod] || [];
            if (!existing.includes(dAct)) {
                updated[dMod] = [...existing, dAct];
            }
        });
    } else {
        // Remove dependency only if no other active permission still requires it
        deps.forEach(({ module: dMod, action: dAct }) => {
            const stillRequired = Object.entries(PERMISSION_DEPENDENCIES).some(([otherKey, otherDeps]) => {
                if (otherKey === key) return false;
                const [oMod, oAct] = otherKey.split(":");
                const isOtherActive = updated[oMod]?.includes(oAct);
                return isOtherActive && otherDeps.some((d) => d.module === dMod && d.action === dAct);
            });
            if (!stillRequired) {
                updated[dMod] = (updated[dMod] || []).filter((a) => a !== dAct);
            }
        });
    }

    return updated;
};

const handleCheckboxChange = (module: string, action: string) => {
    if (isPermissionLocked(module, action)) return; // 🔒 block unchecking locked permissions

    setSelectedPermissions((prev) => {
        const modulePermissions = prev[module] || [];
        const isAdding = !modulePermissions.includes(action);

        const updated = isAdding
            ? { ...prev, [module]: [...modulePermissions, action] }
            : { ...prev, [module]: modulePermissions.filter((a) => a !== action) };

        return applyDependencies(updated, module, action, isAdding);
    });
};


const handleSelectAllChange = (module: string, actions: string[]) => {
    const allSelected = isAllSelected(module, actions);

    setSelectedPermissions((prev) => {
        let updated = { ...prev, [module]: allSelected ? [] : actions };

        // Apply/remove dependencies for each action in this module
        actions.forEach((action) => {
            updated = applyDependencies(updated, module, action, !allSelected);
        });

        return updated;
    });
};



{actions.map((action) => {
    const locked = isPermissionLocked(module, action);
    return (
        <label
            className={`flex items-center gap-2 ${locked ? "cursor-not-allowed opacity-60" : "cursor-pointer"}`}
            key={action}
            title={locked ? "Required by another permission" : undefined}
        >
            <input
                type='checkbox'
                checked={selectedPermissions[module]?.includes(action) || false}
                onChange={() => handleCheckboxChange(module, action)}
                disabled={locked}
            />
            {transfromActionName(action)}
            {locked && <span className='text-xs text-gray-400'>(auto)</span>}
        </label>
    );
})}
